## `__proto__`

```JavaScript
const student = {
  age: 14,
};

const lee = {
  color: `black`,
};

const kim = {
  color: `blue`,
};

lee.__proto__ = student;
kim.__proto__ = student;
console.log(lee.age); //14
```

## `prototype`

```JavaScript
let animal = {
  eats: true,
};

function Rabbit(name) {
  this.name = name;
}

Rabbit.prototype = animal;
let rabbit = new Rabbit(`흰 토끼`);
console.log(rabbit.eats); //true
```

```JavaScript
function Rabbit(name) {
  this.name = name;
}
Rabbit.prototype = {
  jumps: true,
};

let rabbit = new Rabbit(`흰 토끼`);

console.log(rabbit.jumps); //true
```
